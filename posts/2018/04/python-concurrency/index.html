python多线程多进程

因为GIL的原因，多线程只能并发的跑，不能并行，多进程可多核并行

import threading
threads = []
for i in range(4):
    thread = threading.Thread(target=task, args=(param,))
    thread.start()
    threads.append(thread)
for thread in threads:
    thread.join()
